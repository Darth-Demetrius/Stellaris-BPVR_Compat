district_ssh_horizon_brain_server = {
	base_buildtime = @hb_district_buildtime
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	inline_script = { script = districts/BPV_district_slots  GOVERNMENT = slot_ssh_horizon_brain_server_01 SLOT1 = slot_ssh_horizon_brain_server_02 SLOT2 = slot_ssh_horizon_brain_server_02 SLOT3 = slot_ssh_horizon_brain_server_02 SLOT4 = slot_ssh_horizon_brain_server_02 SLOT5 = slot_ssh_horizon_brain_server_02 SLOT6 = slot_ssh_horizon_brain_server_02 SLOT7 = slot_ssh_horizon_brain_server_02 SLOT8 = slot_ssh_horizon_brain_server_02 SLOT9 = slot_ssh_horizon_brain_server_02 SLOT10 = slot_ssh_horizon_brain_server_02 }

	show_on_uncolonized = {
		exists = from
		from = {
			NOT = {
				is_wilderness_empire = yes
			}
		}
		uses_district_set = horizon_brain_world
	}

	potential = {
		exists = owner
		owner = {
			has_technology = ssh_tech_vr # Required
			NOT = { is_wilderness_empire = yes }
		}
		uses_district_set = horizon_brain_world
		NAND = {
			is_capital = yes
			owner = {
				has_menace_perk = menp_behemoth_mind_meld
			}
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_ssh_horizon_brain_grid
		district_mindlink
	}

	resources = {
		category = planet_districts
		cost = {
			alloys = @hb_alloy_cost
		}
		upkeep = {
			energy = @hb_upkeep
		}
	}

	# Add Admin + User/VR Drone Jobs
	inline_script = {
		script = districts/ssh_district_horizon_brain_jobs_regular
		ADMIN_AMOUNT = 200
		TWO_ADMINS_AMOUNT = 100
		USER_AMOUNT = 1000
		#AMOUNT = @suprashell_district_jobs
	}
	inline_script = {
		script = districts/ssh_district_horizon_brain_jobs_gestalt
		VR_DRONE_AMOUNT = 400
	}

	planet_modifier = {
		planet_housing_add = 1500
	}

	ai_estimate_without_unemployment = yes
	additional_ai_weight = 200
	ai_weight_coefficient = 1.2
}
